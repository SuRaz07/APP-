class FlirtingScreen extends StatelessWidget {
  final List<String> flirtingTexts;
  final List<String> favoriteFlirtingTexts;
  final Function(String) toggleFavorite;

  FlirtingScreen({
    required this.flirtingTexts,
    required this.favoriteFlirtingTexts,
    required this.toggleFavorite,
  });

  @override
  Widget build(BuildContext context) {
    return ListView.builder(
      itemCount: flirtingTexts.length,
      itemBuilder: (context, index) {
        String text = flirtingTexts[index];
        bool isFavorite = favoriteFlirtingTexts.contains(text);
        return ListTile(
          title: Text(text),
          trailing: IconButton(
            icon: Icon(
              isFavorite ? Icons.favorite : Icons.favorite_border,
              color: isFavorite ? Colors.red : null,
            ),
            onPressed: () {
              toggleFavorite(text);
            },
          ),
        );
      },
    );
  }
}

class QuotesScreen extends StatelessWidget {
  final List<String> quotes;
  final List<String> favoriteQuotes;
  final Function(String) toggleFavorite;

  QuotesScreen({
    required this.quotes,
    required this.favoriteQuotes,
    required this.toggleFavorite,
  });

  @override
  Widget build(BuildContext context) {
    return ListView.builder(
      itemCount: quotes.length,
      itemBuilder: (context, index) {
        String quote = quotes[index];
        bool isFavorite = favoriteQuotes.contains(quote);
        return ListTile(
          title: Text(quote),
          trailing: IconButton(
            icon: Icon(
              isFavorite ? Icons.favorite : Icons.favorite_border,
              color: isFavorite ? Colors.red : null,
            ),
            onPressed: () {
              toggleFavorite(quote);
            },
          ),
        );
      },
    );
  }
}

class FavoriteScreen extends StatefulWidget {
  final List<String> favoriteFlirtingTexts;
  final List<String> favoriteQuotes;
  final Function(String, List<String>) toggleFavorite;

  FavoriteScreen({
    required this.favoriteFlirtingTexts,
    required this.favoriteQuotes,
    required this.toggleFavorite,
  });

  @override
  _FavoriteScreenState createState() => _FavoriteScreenState();
}

class _FavoriteScreenState extends State<FavoriteScreen> {
  String selectedType = 'flirting';

  @override
  Widget build(BuildContext context) {
    List<String> currentList = selectedType == 'flirting'
        ? widget.favoriteFlirtingTexts
        : widget.favoriteQuotes;

    return Column(
      children: [
        SizedBox(height: 20), // Added space between the nav bar and buttons
        Row(
          mainAxisAlignment: MainAxisAlignment.center,
          children: [
            Container(
              decoration: BoxDecoration(
                borderRadius: BorderRadius.circular(20),
                boxShadow: selectedType == 'flirting'
                    ? [
                        BoxShadow(
                          color: Color.fromARGB(180, 0, 0, 255),
                          blurRadius: 1,
                          spreadRadius: 1,
                        )
                      ]
                    : [],
              ),
              child: ElevatedButton(
                onPressed: () {
                  setState(() {
                    selectedType = 'flirting';
                  });
                },
                child: Text('Flirting Favorites'),
              ),
            ),
            SizedBox(width: 10),
            Container(
              decoration: BoxDecoration(
                borderRadius: BorderRadius.circular(20),
                boxShadow: selectedType == 'quotes'
                    ? [
                        BoxShadow(
                          color: Color.fromARGB(180, 0, 0, 255),
                          blurRadius: 1,
                          spreadRadius: 1,
                        )
                      ]
                    : [],
              ),
              child: ElevatedButton(
                onPressed: () {
                  setState(() {
                    selectedType = 'quotes';
                  });
                },
                child: Text('Quotes Favorites'),
              ),
            ),
          ],
        ),
        Expanded(
          child: currentList.isEmpty
              ? Center(child: Text('No favorites added'))
              : ListView.builder(
                  itemCount: currentList.length,
                  itemBuilder: (context, index) {
                    String item = currentList[index];
                    return ListTile(
                      title: Text(item),
                      trailing: IconButton(
                        icon: Icon(Icons.delete),
                        onPressed: () {
                          widget.toggleFavorite(item, currentList);
                        },
                      ),
                    );
                  },
                ),
        ),
      ],
    );
  }
}
